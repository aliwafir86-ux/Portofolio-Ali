// Menangani pengiriman form tanpa pindah halaman const form = document.getElementById('email-form'); const btnSend = form.querySelector('.send-btn'); form.addEventListener('submit', async function(e) { e.preventDefault(); // Mencegah reload halaman // Ubah teks tombol saat loading const originalText = btnSend.innerText; btnSend.innerText = "Mengirim..."; btnSend.disabled = true; const formData = new FormData(form); try { const response = await fetch(form.action, { method: 'POST', body: formData, headers: { 'Accept': 'application/json' } }); if (response.ok) { // Notifikasi Sukses form.innerHTML = `
✅

Pesan Terkirim!

Terima kasih, Ali akan segera menghubungimu.

`; // Tutup modal otomatis setelah 3 detik setTimeout(() => { tutupModal(); }, 3000); } else { alert("Oops! Terjadi kesalahan, coba lagi nanti."); btnSend.innerText = originalText; btnSend.disabled = false; } } catch (error) { alert("Gagal terhubung ke server."); btnSend.innerText = originalText; btnSend.disabled = false; } });
